program fibonnaciIterativo:
 var int: n, x;
int func fib(int n) { 
  var int: a, b, c, i; 
  a = 0;
  b = 1;
  c = 1;
  if( n == 0) {
    return (a); 
  };

  for (i = 2 to n ) 
  { 
     c = a + b; 
     a = b; 
     b = c; 
  }; 
  return (b); 
}

main(){
  n=10;
  x= fib(fib(3)) + 1;
  write(x);
}
